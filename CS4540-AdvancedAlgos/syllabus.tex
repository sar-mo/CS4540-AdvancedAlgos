\subsection*{Summary}
    This course studies advanced algorithms beyond the standard curriculum in CS3510 and CS3511. It aims to develop a deeper understanding of approaches used in current research trends. The course is divided into 5-7 modules, introducing graduate material and providing a good transition for those planning to pursue graduate studies in theory.

\section*{Syllabus (Tentative)}
    \begin{enumerate}[label=\Roman*.]
        \item Stable Marriages:
        \item Randomized Algorithms
        \begin{itemize}
            \item Universal Hash Families
            \item k-wise independence, linearity of expectation
            \item Min-cut
        \end{itemize}
        \item On-line Algorithms
        \begin{itemize}
            \item Intro to on-line algorithms: page
            \item Randomized paging
            \item k-server problem
            \item Ski-rental
        \end{itemize}
        \item Tail inequalities
        \item Approximation Algorithms
        \begin{itemize}
            \item Vertex cover, traveling salesman, set cover
            \item Knapsack and PTASs
        \end{itemize}
        \item Computational Geometry
        \begin{itemize}
            \item Convex Hulls
            \item Voronoi regions
            \item Delaunay triangulations
        \end{itemize}
        \item Optional topics
        \begin{itemize}
            \item Random Walks and Markov Chain Fundamentals
            \item Optimization
            \item Shaders
            \item FFT
        \end{itemize}
    \end{enumerate}
